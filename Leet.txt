  public int maxSubArray(int[] nums) {
        int n = nums.length;
        int maxa = nums[0];
        int max = nums[0];

        for(int i = 1; i < n; i++) {
            if(max + nums[i] < nums[i]) {
                max = nums[i];
            } else {
                max = max + nums[i];
            }

            if(max > maxa) {
                maxa = max;
            }
        }
        return maxa;
    }
